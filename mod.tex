 \dfrac{d^2y}{d^2x}+x\brak{\dfrac{dy}{dx}}^2=2x^2\log\brak{\dfrac{d^2y}{dx^2}}
\end{align*}

\begin{align*}
 \int{\sqrt{3 - 2x - x^2}}{dx}

$\begin{vmatrix}
3a & -a + b & -a + c \\
 -b + a & 3b & -b + c \\
-c + a & -c + b & 3c

\item Let $f : N \rightarrow Y$ be a function defined as 
\begin{equation}
 f(x) = 4x + 3,
\end{equation}

x^3\frac{d^2y}{dx^2} = \brak{x\frac{dy}{dx}-y}^2.

 5 & 1 & 0 \\
 0 & 1 & 3 \\
